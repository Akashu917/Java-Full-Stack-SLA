1.Add a button click event, Show input value when typing, Add event listener to a button:

Ans:

<input type="text" id="liveInput" placeholder="Type here">
<button id="btn">Submit</button>
<p id="result"></p>

<script>
  let input = document.getElementById("liveInput");
  let btn = document.getElementById("btn");
  let result = document.getElementById("result");

  // Show typing
  input.addEventListener("input", () => {
    result.innerText = input.value;
  });

  // Button click
  btn.addEventListener("click", () => {
    alert("You entered: " + input.value);
  });
</script>
