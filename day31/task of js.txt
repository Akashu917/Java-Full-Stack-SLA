1.Function that Accepts a Callback:

Ans:

function processOrder(order, callback) {
  console.log("Processing order:", order);
  callback(order);
}

// Callback function
function orderComplete(order) {
  console.log("Order completed:", order);
}

// Call
processOrder("Shoes", orderComplete);


2.Create a Promise (Resolve / Reject):

Ans:

const myPromise = new Promise((resolve, reject) => {
  let success = true;

  if (success) {
    resolve("Payment successful");
  } else {
    reject("Payment failed");
  }
});

3.Handle Promise using .then() and .catch():

Ans:

myPromise
  .then(result => {
    console.log(result); // if resolved
  })
  .catch(error => {
    console.log(error); // if rejected
  });

Example:

function checkPayment(status) {
  return new Promise((resolve, reject) => {
    if (status) {
      resolve("Order placed successfully");
    } else {
      reject("Order failed");
    }
  });
}

checkPayment(true)
  .then(msg => console.log(msg))
  .catch(err => console.log(err));
